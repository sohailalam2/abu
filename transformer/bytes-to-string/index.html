Usage #

ts
// Hello World! encoded as Uint8Array
const input = Uint8Array.from([72, 101, 108, 108, 111, 32, 87, 111, 114, 108, 100, 33]);

const result = new Uint8ArrayToString().convert(input);
expect(result).toEqual('Hello World!'); // ✅
